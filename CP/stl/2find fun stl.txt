#include <algorithm>
using namespace std;



int main()
{

	freopen("input.txt", "r", stdin);
	freopen("output.txt", "w", stdout);

	int arr[] = {5 , 4 , 2, 6, 7 , 8, 1};
	int key = 7 ; // the element that you want to search
// size of the array
	int size = sizeof(arr) / sizeof(int);


// now we use find()

	auto it = find(arr , arr + size , key); // now find in from the array , those are the index range

	cout << it << endl; // this returns an address of the number
	// to get index we must subtract it with base address
	// in array if you subtract two pointers then you get the number of locations in between
	cout << arr << endl;

	// if number not in array then it returns last index to avoid that we use 'if'
	if (it - arr == size) {
		cout << "not present" << endl;
	}
	else {
		cout << "index is " << it - arr << endl;

	}

}
