#include <bits/stdc++.h>
#include <algorithm>
using namespace std;

template <class T> // or typename T
//int search(int arr[] , int n , int key)
int search(T arr[] , int n , T key)
{
	int i ;
	for (i = 0 ; i < n ; i++) {

		if (arr[i] == key) {
			return i ;
		}

	}
	return 0 ;
}



int main()
{

	freopen("input.txt", "r", stdin);
	freopen("output.txt", "w", stdout);

// it is always suggestable to use templates in your program so that it can run for generic inputs - char/int...
	int arr[] = {5 , 4 , 2, 6, 7 , 8, 1};
	int key = 7 ; // the element that you want to search
// size of the array
	int size = sizeof(arr) / sizeof(int);
	cout << "the position is " << search(arr , size , key ) << endl;

	float arr_[] = {1.2, 3.4, 6.7, 8.9, 7.5, 0.6};
	float key_ = 0.6;
	int size_ = sizeof(arr_) / sizeof(int);
	cout << "the position is " << search(arr_ , size_ , key_ ) << endl;
}
