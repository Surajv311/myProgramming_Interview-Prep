#include <bits/stdc++.h>
#include <algorithm>
using namespace std;

int fib(int n) {
	int ans;
	if (n == 1 or n == 0) {
		return n;
	}
	else {
		ans = fib(n - 1) + fib(n - 2);

		return ans;
	}

}
