#include <bits/stdc++.h>
#include <algorithm>
using namespace std;

int fib(int n, int dp[]) {
	// now using dynamic programming
	int ans;
	if (n == 1 or n == 0) {
		return n;
	}
	if (dp[n] != 0) {
		return dp[n];
	}

	ans = fib(n - 1, dp) + fib(n - 2, dp);

	return dp[n] = ans;


}
