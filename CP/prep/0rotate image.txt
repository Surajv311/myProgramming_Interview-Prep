void rotateFunction(int arr[][100], int n ) {
// first we need to swap elements in each row
	int row , startCol , endCol, col;
	startCol = 0 ;
	endCol = n - 1 ;
	for (row = 0 ; row < n ; row++) {
		while (startCol > endCol) {
			swap(arr[row][startCol], arr[row][endCol]);
			startCol++;
			endCol--;
		}
	}
	// now taking transpose
	for (row = 0 ; row < n  ; row++)
	{
		for (col = 0 ; col < n ; col++)
		{
			if (row < col)
			{
				swap(arr[row][col], arr[col][row]);
			}
		}

	}
}
void display(int arr[][100], int n) {
	for (int i = 0 ; i < n ; i++) {
		for (int j = 0; j < n; j++) {
			cout << arr[i][j] << " ";
		}
		cout << endl;
	}
}
