#include <bits/stdc++.h>
#include <algorithm>
using namespace std;


bool comparator(int a , int b) {

	return a > b;
}
int main()
{

	freopen("input.txt", "r", stdin);
	freopen("output.txt", "w", stdout);

	int arr[] = {5 , 4 , 2, 6, 7 , 7, 7, 8, 1};
// to sort the function
	int size = sizeof(arr) / sizeof(int);
	sort(arr , arr + size);

	for (int i = 0 ; i < size ; i++)
		cout << arr[i] << " " ;

	cout << endl;
	// now suppose I want to compare and reverse the order of sorting

	// we use comparators

	sort(arr , arr + size , comparator);
	for (int i = 0 ; i < size ; i++)
		cout << arr[i] << " ";


}
