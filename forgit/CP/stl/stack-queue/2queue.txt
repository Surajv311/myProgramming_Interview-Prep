#include <bits/stdc++.h>
#include <algorithm>
#include <vector>
#include <queue>
using namespace std;

int main()
{

	freopen("input.txt", "r", stdin);
	freopen("output.txt", "w", stdout);

	queue<int>q; // using vectors

	int n ; // size of the stack
	cin >> n;
	for (int i = 0 ; i < n; i++) {
		q.push(i);
	}

	while (!q.empty()) {

		cout << q.front();
		q.pop();
	}
}
