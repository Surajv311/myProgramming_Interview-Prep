#include <bits/stdc++.h>
#include <algorithm>
using namespace std;
void bubble_sort(int arr[], int n ) {

// now we have to sort the array
	int i , j ;
	for (i = 0; i < n; i++) {

		for (j = 0; j < n - 1 - i; j++) {
			if (arr[j] > arr[j + 1]) {
				swap(arr[j], arr[j + 1]);
			}

		}
	}
}

void display(int arr[], int n) {
	for (int i = 0; i < n; i++) {
		cout << arr[i] << " ";
	}
}
int main()
{

#ifndef ONLINE_JUDGE

#endif
	int n ;
// size of array
	cin >> n ;
	int arr[n];
	int i;
	for (i = 0; i < n; i++) {
		cin >> arr[i];
	}

	bubble_sort(arr, n);
	display(arr, n);
}
