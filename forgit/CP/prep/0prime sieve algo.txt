#include <bits/stdc++.h>
using namespace std;
int primeSeiveAlgo(int* array) // you may even pass int[]
{
	// first marking all odd numbers as prime
	// we are not considering even numbers as they are not prime , except 2 .
	for (int i = 3; i <= 10000; i += 2)
	{
		array[i] = 1;
	}
	for (int i = 3; i <= 10000; i += 2)
	{
		if (array[i] == 1)
		{
			for (int j = i * i ; j <= 10000 ; j = j + i)
			{
				array[j] = 0;
			}
		}
	}
// special case of '2' as prime number
	array[2] = 1;
	array[1] = array[0] = 0;
}

int main()
{

#ifndef ONLINE_JUDGE
	freopen("input.txt", "r", stdin);
	freopen("output.txt", "w", stdout);
#endif
	int prime[10000] = {0};
	int n ;
	cin >> n ;
// prime seive algo to find prime numbers
	primeSeiveAlgo(prime);
	for (int i = 0 ; i <= n; i++)
	{
		if (prime[i] == 1) {
			cout << i << " " ;
		}
	}

}
