#include <bits/stdc++.h>
using namespace std;
int gcdByEuclidAlgorithm(int a_ , int b_)
{
	return b_ == 0 ? a_ : gcdByEuclidAlgorithm(b_, a_ % b_) ;
	// means if b_==0 return a_ else return...
	//IF YOU WANT TO FIND LCM THEN : GCD*LCM = A*B
}
