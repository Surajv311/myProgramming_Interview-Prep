#include <bits/stdc++.h>
using namespace std;
int trailingZeroes(int num)
{
	int ans = 0;
	int x = 5 ;
	while ((num / x) > 0)
	{
		ans += (num / x) ;
		x = x * 5 ;
	}
	return ans ;
}
